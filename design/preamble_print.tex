\newcommand{\dichapter}[3]{%
  \cleardoublepage
  \begingroup\raggedright
  {\displaysemi\large\MakeUppercase{#1}\par}%
  \vspace{3pt}{\rule{\linewidth}{1.5pt}}\par
  \vspace{12pt}{\displayfont\fontsize{25}{29}\selectfont #2\par}%
  \vspace{6pt}{\itshape\color{mid}\large #3\par}%
  \vspace{3pt}{\color{mid}\rule{\linewidth}{0.5pt}}\par
  \endgroup\vspace{15pt}
  \addcontentsline{toc}{chapter}{\texorpdfstring{#1 \textbar\ #2}{#1 | #2}}%
}

% Section banner: solid black bar, white text (color arg ignored)
\newcommand{\dibanner}[2]{%
  \par\addvspace{13pt}%
  {\setlength{\fboxsep}{6pt}\noindent\colorbox{ink}{%
     \begin{minipage}{\dimexpr\linewidth-2\fboxsep\relax}%
       \displaysemi\color{white}\normalsize\strut\MakeUppercase{#2}\strut%
     \end{minipage}}}%
  \par\addvspace{8pt}%
}

% Boxed equation: black frame, white fill, black math
\newcommand{\eqbox}[1]{%
  \par\addvspace{11pt}\begin{center}%
  {\setlength{\fboxsep}{9pt}\setlength{\fboxrule}{0.8pt}%
   \fcolorbox{ink}{white}{\begin{minipage}{0.86\linewidth}\centering\large #1\end{minipage}}}%
  \end{center}\addvspace{11pt}}
