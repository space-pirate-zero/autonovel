% ---- Chapter opener (called from body via raw \dichapter) ----
\newcommand{\dichapter}[3]{%
  \clearpage
  \begingroup\raggedright
  {\displaysemi\color{cyan}\large\MakeUppercase{#1}\par}%
  \vspace{3pt}{\color{pink}\rule{\linewidth}{2.5pt}}\par
  \vspace{14pt}
  {\displayfont\color{pink}\fontsize{30}{34}\selectfont #2\par}%
  \vspace{8pt}
  {\itshape\color{cyan}\large #3\par}%
  \vspace{4pt}{\color{cyan!70}\rule{\linewidth}{0.6pt}}\par
  \endgroup
  \vspace{18pt}
  \addcontentsline{toc}{chapter}{\texorpdfstring{#1 \textbar\ #2}{#1 | #2}}%
}

% ---- Colored section banner (emitted by sections.lua) ----
\newcommand{\dibanner}[2]{%
  \par\addvspace{16pt}%
  {\setlength{\fboxsep}{7pt}%
   \noindent\colorbox{#1}{%
     \begin{minipage}{\dimexpr\linewidth-2\fboxsep\relax}%
       \displaysemi\color{void}\large\strut\MakeUppercase{#2}\strut%
     \end{minipage}}}%
  \par\addvspace{9pt}%
}
